
SELECT ENAME FROM EMPLOYEE WHERE SALARY = (SELECT SALARY FROM EMPLOYEE GROUP BY SALARY HAVING COUNT(SALARY)>1);



SELECT DNO, ENAME, DOJ FROM EMPLOYEE WHERE DOJ IN (SELECT MIN(DOJ) FROM EMPLOYEE GROUP BY DNO);


SELECT TO_CHAR(DOJ, 'MONTH') AS JOIN_MONTH FROM EMPLOYEE GROUP BY TO_CHAR(DOJ, 'MONTH') HAVING COUNT(TO_CHAR(DOJ,'MONTH')) = ( SELECT MAX(COUNT(TO_CHAR(DOJ,'MONTH'))) FROM EMPLOYEE GROUP BY TO_CHAR(DOJ, 'MONTH') );
